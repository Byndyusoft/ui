### 6. onChange
```jsx

  onChange: (isIntersecting, entry) => {
      console.log('useIntersectionObserver onChange', isIntersecting, entry)
      if (isIntersecting) {
          alert(`isIntersecting: ${isIntersecting}`);
      }
  }
});
```
